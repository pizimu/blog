//js代码
var loading = function(){
    var count = 0;
    return function(status){
        var loadingHtml = '
';            
        if(window.top.$('#loading').length === 0){
            window.top.$('body').append(loadingHtml);
        }
        status ? count++ :count--;
        if(count < 1){
            count = 0;
            window.top.$('#loading').hide();
        }else{
            window.top.$('#loading').show();
        }
        return true;
    };
};
