实例代码：点击查看 

/*
 *前端定时任务
 * 不支持同一时间多个任务
 * 如需使用服务器时间，请修改 serverTime方法，以获取服务器时间
*/
timingTasks({
  /*
   *任务列表 list
   *参数
   *  time:执行请求时间
   *  可添加其它参数，做为调用callback方法时的标示，如
   *  	type:'begin',
   *  	id:'time1'
  */
  tasks:[],
  /*
   *回调函数
   *当到达时间时，执行回调函数
   *参数
   * obj:tasks中，time在当前时间的选项
  */
  callback:function(obj){},
  /*
   *周末不执行
  */
  ridWeekend:true,
  /*
   *是否每日任务，可与ridWeekend混合使用
  */
daily:true
});
