$.loading

用法

// $.loading(status)
// loading 永远只有一个。每次打开，与关闭，不影响其它 loading。
// 参数: 
//   status: Boolean。为 true值 或 空值 时，打开loading ;为 false值 时，关闭loading

// 打开
	$.loading();
	$.loading(true);
// 关闭
	$.loading(false);
	$.loading(0);

$.fn.loading

用法

// $.fn.loading(onoff, cls)
// 在当前元素加载loading
// 参数: 
//   onoff: Boolean。为 true值 或 空值 时，打开loading ;为 false值 时，关闭loading
//   cls: String。 为 loading 添加的类。自带值：mini(默认) | big。 可自定义添加

// 打开
	$(selector).loading();
	$(selector).loading(true, 'big');
// 关闭
	$(selector).loading(false);
	$(selector).loading(0);
